}

}

}

}

+ distanciaEntreCoordenadas(c1: Coordenada, c2: Coordenada): int

}

- columna
+ esPosibleConstruir(): boolean
}

class PasarelaMeta{}

- columna
+ esPosibleConstruir(): boolean
}

- columna

}
