@startuml
abstract Enemigo {
- velocidad: int
- dañoCausado: int
- energía: int
- creditosOtorgados: int
- pasarela: Pasarela
- coordenada: Coordenada
+ cantidadCreditosOtorgados(): int
+ estaVivo(): boolean
}

class Hormiga {
+ cantidadCreditosOtorgados(): int
}

class Araña {
+ cantidadCreditosOtorgados(): int
}

class Juego {
- mapa: Mapa
- turno: Turno
+ avanzarTurno()
}

class Inicializador {
+ agregarJugador()
+ iniciarPartida()
}

abstract Defensa {
- nombre
- costo
- tiempoDeConstruccion
- rangoDeAtaque
- daño
- tierra: Tierra
- coordenada: Coordenada
+ atacarEnemigo(enemigo: Enemigo)
+ construir(torre:Torre)
}

class TorreBlanca {
+ atacarEnemigo(enemigo: Enemigo)
+ construir(torre:Torre)
}

class TorrePlateada {
+ atacarEnemigo(enemigo: Enemigo)
+ construir(torre:Torre)
}

class Jugador {
- nombre
- cantidadDeCreditos: int
- puntosDeVida: int
+ descontarCreditos()
+ agregarCreditos()
+ generarDefensa()
+ descontarPuntosDeVida()
}


interface EstadoDefensa {
+atacarEnemigo(e: Enemigo, D: Defensa)
}

class EnConstruccion {
+atacarEnemigo(e: Enemigo, D: Defensa)
}

class Terminada {
+atacarEnemigo(e: Enemigo, D: Defensa)
}


interface EstadoEnemigo {
+ mover(enemigo:Enemigo)
}

class Vivo {
+ mover(enemigo:Enemigo)
}

class Muerto {
+ mover(enemigo:Enemigo)
}

EstadoEnemigo <|.. Vivo
EstadoEnemigo <|.. Muerto

EstadoDefensa <|.. EnConstruccion
EstadoDefensa <|.. Terminada

Defensa <|-- TorreBlanca
Defensa <|-- TorrePlateada



Enemigo <|-- Hormiga
Enemigo <|-- Araña

Juego <- Inicializador
Juego "1" --> "0..*" Enemigo
Jugador <- Juego
Jugador "1" --> "0..*" Defensa
Defensa .> Enemigo
Defensa --> EstadoDefensa
Enemigo --> EstadoEnemigo
@enduml
